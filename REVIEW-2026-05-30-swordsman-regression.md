    doc_val = 26 if n == 10 else (36 if n == 12 else (42 if n == 13 else 33))

defense = 1000

    print(f"{skill}: 精确={exact:.0f}, 文档={doc_vals[skill]}, 差={diff:.0f}")

total_exact = sum(base_atk * p * strength_coef * defense_coef * crit_coef for p in [4.20, 3.80, 3.00])
