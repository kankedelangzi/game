```python
# === 暴击期望公式（DNF 70版本标准） ===
# 暴击期望系数 = (1 - 暴击率) + 暴击率 × 1.5 = 1 + 0.5 × 暴击率
# 收益比 = (1 + 0.5 × crit_after) / (1 + 0.5 × crit_before) - 1

# 狂战士: 55% → 58%
crit_before_bs = 0.55
crit_after_bs = 0.58
exp_before_bs = 1 + 0.5 * crit_before_bs  # 1.275
exp_after_bs = 1 + 0.5 * crit_after_bs    # 1.290
crit_bonus_bs = exp_after_bs / exp_before_bs - 1  # 1.18%

# 剑魂: 50% → 53%
crit_before_sw = 0.50
crit_after_sw = 0.53
exp_before_sw = 1 + 0.5 * crit_before_sw  # 1.250
exp_after_sw = 1 + 0.5 * crit_after_sw    # 1.265
crit_bonus_sw = exp_after_sw / exp_before_sw - 1  # 1.20%

# 狂战士固伤综合
ind_bonus = (1 + 1370/250) / (1 + 1250/250) - 1  # 8.0%
fixed_total = (1 + ind_bonus) * (1 + crit_bonus_bs) - 1  # 9.27%

# 狂战士百分比综合
str_bonus = (1 + 1038/250) / (1 + 728/250) - 1  # 31.70%
phy_bonus = (1 + 2110/2500) / (1 + 2000/2500) - 1  # 2.44%
percent_total = (1 + str_bonus) * (1 + phy_bonus) * (1 + crit_bonus_bs) - 1  # 36.50%

# 剑魂百分比综合
str_bonus_sw = (1 + 910/250) / (1 + 600/250) - 1  # 36.47%
phy_bonus_sw = (1 + 2710/2500) / (1 + 2600/2500) - 1  # 2.16%
percent_total_sw = (1 + str_bonus_sw) * (1 + phy_bonus_sw) * (1 + crit_bonus_sw) - 1  # 41.09%
```
