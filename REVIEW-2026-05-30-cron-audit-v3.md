## 七、附录：Python独立验算

```python
# === 武器强化系数验证 ===
print("=== 武器强化系数验证 ===")
official = {10: 30, 11: 33, 12: 36, 13: 42}
doc = {10: 26, 11: 33, 12: 36, 13: 42}
for n in official:
    status = "✅" if official[n] == doc[n] else "❌"
    print(f"+{n}: 官方={official[n]}%, 文档={doc[n]}% {status}")

# === 伤害计算精确验算 ===
print("\n=== 伤害计算精确验算 ===")
base_atk = 2600
strength = 600
strength_coef = 1 + strength / 250  # 3.4
defense = 1000
defense_coef = 300 / (300 + defense)  # 0.2307692308
crit_coef = 1.25

for skill, pct, doc_val in [("裂波斩", 4.20, 10721), ("拔刀斩", 3.80, 9699), ("上挑", 3.00, 7658)]:
    exact = base_atk * pct * strength_coef * defense_coef * crit_coef
    diff = doc_val - exact
    print(f"{skill}: 精确={exact:.0f}, 文档={doc_val}, 差={diff:.0f}")

# === CC套收益验算 ===
```
