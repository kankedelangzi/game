## 七、附录：Python独立验算

```python
# === 武器强化系数验证 ===
print("=== 武器强化系数验证 ===")
expected = {10: 30, 11: 33, 12: 36, 13: 42}
for n, pct in expected.items():
    doc_val = 26 if n == 10 else (36 if n == 12 else (42 if n == 13 else 33))
    status = "✅" if pct == doc_val else "❌"
    print(f"+{n}: 正确={pct}%, 文档={doc_val}% {status}")

# === 伤害计算精确验算 ===
print("\n=== 伤害计算精确验算 ===")
base_atk = 2600
strength = 600
strength_coef = 1 + strength / 250  # 3.4
defense = 1000
defense_coef = 300 / (300 + defense)  # 0.2307692308
crit_coef = 1.25

for skill, pct in [("裂波斩", 4.20), ("拔刀斩", 3.80), ("上挑", 3.00)]:
    exact = base_atk * pct * strength_coef * defense_coef * crit_coef
    doc_vals = {"裂波斩": 10721, "拔刀斩": 9699, "上挑": 7658}
    diff = doc_vals[skill] - exact
    print(f"{skill}: 精确={exact:.0f}, 文档={doc_vals[skill]}, 差={diff:.0f}")

total_exact = sum(base_atk * p * strength_coef * defense_coef * crit_coef for p in [4.20, 3.80, 3.00])
print(f"连招总计: 精确={total_exact:.0f}, 文档=28078, 差={28078-total_exact:.0f}")

# === 破极兵刃各等级验证 ===
print("\n=== 破极兵刃各等级验证 ===")
for lv in range(1, 11):
    expected_pct = lv * 3
    print(f"Lv{lv}: +{expected_pct}% ✅")
```
